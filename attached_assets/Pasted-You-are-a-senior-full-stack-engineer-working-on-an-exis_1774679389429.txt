### 3. OWASP Tagging + Severity Mapping

Create a mapping utility:

```ts
function mapToOWASP(issue): { category, severity } {
  if (issue.type.includes("SQL") || issue.type.includes("XSS")) {
    return { owaspCategory: "A03: Injection", severity: "Critical" };
  }
  if (issue.type.includes("missing label") || issue.type.includes("alt")) {
    return { owaspCategory: "A11: Accessibility", severity: "Medium" };
  }
  return { owaspCategory: "General", severity: "Low" };
}
```

Apply during issue creation.

### 4. Health Score Calculation

Add function:

```ts
function calculateHealthScore(issues: Issue[]): number {
  let score = 100;

  for (const issue of issues) {
    if (issue.severity === "Critical") score -= 10;
    if (issue.severity === "High") score -= 5;
    if (issue.severity === "Medium") score -= 2;
    if (issue.severity === "Low") score -= 1;
  }

  return Math.max(score, 0);
}
```

Attach to report.json:

```ts
report.healthScore = calculateHealthScore(issues);
```
